Risk Modeling – Black-Scholes

Call option price:

$$C = S_0 N(d_1) - K e^{-rT} N(d_2)$$
$$d_1 = \frac{\ln(S_0/K) + (r + \sigma^2/2)T}{\sigma\sqrt{T}}$$
$$d_2 = d_1 - \sigma\sqrt{T}$$

import pandas as pd
import numpy as np
from scipy.stats import norm

def black_scholes(S, K, T, r, sigma, type="call"):
    d1 = (np.log(S/K) + (r + sigma**2/2)*T) / (sigma*np.sqrt(T))
    d2 = d1 - sigma*np.sqrt(T)
    
    if type == "call":
        price = S*norm.cdf(d1) - K*np.exp(-r*T)*norm.cdf(d2)
    elif type == "put":
        price = K*np.exp(-r*T)*norm.cdf(-d2) - S*norm.cdf(-d1)
    
    return price

# Live pricing example
option_price = black_scholes(S=4320, K=4300, T=0.25, r=0.0525, sigma=0.18)
print(f"Call option fair value: ${option_price:.2f}")
